package nu.te4.testerjunit;

/**
 *
 * @author Elev
 */
public class MyMath2 {
    
        
    public double convertToFarenheit(double c){
        if(c<-273.15){
            throw new IllegalArgumentException("Temperature cant be lower than 0 Kelvin.");
        }            
         double f = 1.8*c + 32;
        return f; 
    }
    
    public int compareTo(int a, int b){
        if(a > b){
            return a - b;
        }
        else if(a == b){
            return a;
        }
        else if(a < b){
            return a - b;
        }
        return 0; 
    }    
}
